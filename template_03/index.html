Приклад HTML коду

Тут пояснення матеріалу. Так виглядають акценти. 

<!-- приклад HTML коду -->
<h1 id="title">Hello, world!</h1>

Приклад CSS коду

Для правильної підсвітки коду бажано додавати до тегу code класи: language-html, language-css або language-javascript. 

/* приклад CSS коду */
h1 { 
  font-family: monospace;
  font-weight: normal;
}

⚡️ Важливо: Так можна наголосити на важливій інформації.

✨ Примітка: Так можна написати додаткову інформацію.

const pageTitle = document.querySelector("#title");

pageTitle.textContent = "My first webpage";
pageTitle.style.color = "seagreen";

Приклад довгого коду

const viewDataButton = document.getElementById("viewData");
const inputArea = document.getElementById("inputArea");
const outputDiv = document.getElementById("output");

function run() {
  const jsonData = inputArea.value;
  try {
    const data = JSON.parse(jsonData);

    outputDiv.innerHTML = "";

    const pageTitle = document.createElement("h2");
    pageTitle.textContent = `Page of ${data.name} ${data.surname}`;
    outputDiv.appendChild(pageTitle);

    const nameLine = document.createElement("p");
    nameLine.textContent = `Name: ${data.name}`;
    outputDiv.appendChild(nameLine);

    const surnameLine = document.createElement("p");
    surnameLine.textContent = `Surname: ${data.surname}`;
    outputDiv.appendChild(surnameLine);
  } catch (error) {
    outputDiv.textContent = "Invalid JSON input. Please enter valid JSON data.";
  }
}
